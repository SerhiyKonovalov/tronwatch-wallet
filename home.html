Quick Start Example 
// Initialize TronWatch SDK
                                
import TronWatch from '@tronwatch/sdk';

const wallet = new TronWatch({
  apiKey: 'your-api-key',
  network: 'mainnet'
});

// Get wallet balance
const balance = await wallet.getBalance(
  'TRX_ADDRESS_HERE'
);

console.log(`Balance: ${balance} TRX`);
